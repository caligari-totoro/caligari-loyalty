            // 添加積分

            document.getElementById('add-points-btn').addEventListener('click', function() {

                const phone = document.getElementById('point-phone').value;

                const amount = parseFloat(document.getElementById('point-amount').value);

                const resultDiv = document.getElementById('point-result');

                

                if (!phone || !amount) {

                    resultDiv.innerHTML = '<span style="color: red;">請填寫完整信息</span>';

                    return;

                }

                

                showLoading();

                

                db.collection("members").doc(phone).get()

                    .then((doc) => {

                        if (!doc.exists) {

                            hideLoading();

                            resultDiv.innerHTML = '<span style="color: red;">未找到該會員</span>';

                            return;

                        }

                        

                        const member = doc.data();

                        let pointsToAdd;

                        

                        // 根據會員等級計算積分

                        if (member.tier === 'Curry Lover') {

                            // 每消費$100 = 10分

                            pointsToAdd = Math.floor(amount / 100) * 10;

                        } else {

                            // 每消費$80 = 15分

                            pointsToAdd = Math.floor(amount / 80) * 15;

                        }

                        

                        // 更新會員積分

                        const totalPoints = (member.totalPoints || 0) + pointsToAdd;

                        const availablePoints = (member.availablePoints || 0) + pointsToAdd;

                        

                        // 檢查是否需要升級

                        let tier = member.tier;

                        let tierChanged = false;

                        if (tier === 'Curry Lover' && totalPoints >= 500) {

                            tier = 'Curry Master';

                            tierChanged = true;

                        }

                        

                        // 更新數據庫

                        return db.collection("members").doc(phone).update({

                            totalPoints: totalPoints,

                            availablePoints: availablePoints,

                            tier: tier

                        }).then(() => {

                            hideLoading();

                            resultDiv.innerHTML = `

                                <span style="color: green;">成功添加 ${pointsToAdd} 積分給 ${member.name}</span>

                                ${tierChanged ? '<p style="color: blue; margin-top: 5px;">會員已升級為咖喱大師！</p>' : ''}

                            `;

                            

                            // 清空表單

                            document.getElementById('point-phone').value = '';

                            document.getElementById('point-amount').value = '';

                            

                            // 重新載入會員列表

                            loadMembers();

                            

                            // 如果當前用戶是被修改的會員，更新顯示

                            if (currentUser && currentUser.phone === phone) {

                                getMemberInfo(phone);

                            }

                        });

                    })

                    .catch((error) => {

                        hideLoading();

                        resultDiv.innerHTML = '<span style="color: red;">添加積分時出錯：' + error.message + '</span>';

                    });

            });

        function getMemberInfo(phone) {

                    } else {

                    }

                })

                .catch((error) => {

                    hideLoading();

                    showMessage('獲取會員信息失敗：' + error.message);

        }

        // 載入會員資料到管理員面板

        function loadMembers() {

            showLoading();

            

            db.collection("members").get()

                .then((querySnapshot) => {

                    hideLoading();

                    

                    const membersTable = document.getElementById('members-list');

                    membersTable.innerHTML = '';

                    

                    querySnapshot.forEach((doc) => {

                        const member = doc.data();

                        const row = document.createElement('tr');

                        

                        // 格式化日期

                        let joinDate = '未知';

                        if (member.joinDate) {

                            const date = new Date(member.joinDate);

                            joinDate = `${date.getFullYear()}-${(date.getMonth()+1).toString().padStart(2, '0')}-${date.getDate().toString().padStart(2, '0')}`;

                        }

                        

                        // 轉換會員等級顯示

                        const tierDisplay = member.tier === 'Curry Lover' ? '咖喱愛好者' : '咖喱大師';

                        

                        row.innerHTML = `

                            <td>${member.name || '未設置'}</td>

                            <td>${member.phone || '未設置'}</td>

                            <td>${member.birthday || '未設置'}</td>

                            <td>${tierDisplay}</td>

                            <td>${member.totalPoints || 0}</td>

                            <td>${member.availablePoints || 0}</td>

                            <td>${joinDate}</td>

                        `;

                        

                        membersTable.appendChild(row);

                    });

                })

                .catch((error) => {

                    hideLoading();

                    showMessage('獲取會員列表失敗：' + error.message);

                });

        }

        

        // 標籤切換功能

        function openTab(evt, tabName) {

            var i, tabcontent, tablinks;

            

            tabcontent = document.getElementsByClassName("tabcontent");

            for (i = 0; i < tabcontent.length; i++) {

                tabcontent[i].style.display = "none";

            }

            

            tablinks = document.getElementsByClassName("tablinks");

            for (i = 0; i < tablinks.length; i++) {

                tablinks[i].className = tablinks[i].className.replace(" active", "");

            }

            

            document.getElementById(tabName).style.display = "block";

            evt.currentTarget.className += " active";

        }
